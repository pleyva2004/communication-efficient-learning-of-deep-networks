\title{An effective-ODE (averaging-theory) drift bound for one round of
FederatedAveraging}
\author{Mini-study extending McMahan et al.\ (2017), arXiv:1602.05629}
\date{}

\begin{abstract}
For one round of FederatedAveraging (FedAvg) we idealize each client's $u_k$
local full-batch gradient steps as the gradient flow of its local objective run
for continuous time $T_k=\eta\,u_k$. We prove that the server map---the
weight-averaged composition of these per-client flows---agrees with the gradient
flow of the \emph{global} objective $f$ to second order in the time horizon, with
the leading discrepancy equal to a curvature--gradient covariance across clients:
\[
\Big\|\sum_k\beta_k\Phi_k^{T}(w)-\Phi_f^{T}(w)\Big\|
=\tfrac{T^2}{2}\,\sigma_{HG}+O(T^3),
\qquad
\sigma_{HG}=\norm{\Cov_k(H_k,g_k)},
\]
with an explicit cubic remainder ($C_3=\rho G^2+L^2G$) under a bounded-third-derivative assumption.
The leading $\tfrac{T^2}{2}$ term is independent of how the horizon $T$ is split into local steps, so the per-round drift depends on $(E,B,\eta)$ only through the single invariant $T=\eta E n_k/B$ (the discrete FedAvg map matches the flow up to an $O(T/u_k)$ Euler error, Lemma~\ref{lem:euler}). The heuristic
$\Delta=\eta^2\Cov_k(H_k,g_k)$ of the parent study's deep dive shares the \emph{same} drift object $\Gamma=\Cov_k(H_k,g_k)$; at $E=2$, $B=\infty$ ($T=2\eta$) the flow gives $2\eta^2\Gamma$, twice the discrete two-step heuristic's $\eta^2\Gamma$---the factor $2$ being exactly the Euler error of the $2$-step sketch---and the correct prefactor $T^2/2=(\eta u_k)^2/2$ grows as $u_k^2/2$ relative to the frozen $\eta^2$.
\end{abstract}

\section{Setup}

There are $K$ clients with local objectives $F_k:\R^d\to\R$, weights
$\beta_k\ge 0$, $\sum_k\beta_k=1$, and global objective
\[
f=\sum_{k=1}^K\beta_k F_k .
\]
We write the gradient and Hessian at the broadcast point $w$ as
$g_k=\grad F_k(w)$, $H_k=\grad^2 F_k(w)$, $g=\grad f(w)=\sum_k\beta_k g_k$,
$H=\grad^2 f(w)=\sum_k\beta_k H_k$. (We treat $w\in\R^d$; the matrix-parameter
softmax model of the prototype is the vectorized case $w=\mathrm{vec}(W)$,
$d=\dim W$, under the Frobenius inner product, to which everything applies
verbatim.) Throughout, $\norm{\cdot}$ is the Euclidean norm on vectors and the
induced (operator) norm on the matrix $\grad^2 F_k$ and the $3$-tensor
$\grad^3 F_k$; these are compatible, so $\norm{Ax}\le\norm{A}\,\norm{x}$. For the
matrix-valued $H_k$ and vector-valued $g_k$ we write
$\Cov_k(H_k,g_k):=\mathbb E_k[H_kg_k]-\mathbb E_k[H_k]\,\mathbb E_k[g_k]\in\R^d$,
the covariance taken \emph{through} the matrix--vector product, with
$\mathbb E_k$ the expectation under the weights $\beta_k$.

\begin{assumption}[Smoothness]\label{ass:smooth}
There is a closed ball $\mathcal B\subset\R^d$ containing $w$ and a constant
$\rho<\infty$ such that each $F_k\in C^3(\mathcal B)$ with
$\sup_{v\in\mathcal B}\norm{\grad^3 F_k(v)}\le\rho$ and
$\sup_{v\in\mathcal B}\norm{\grad F_k(v)}\le G$,
$\sup_{v\in\mathcal B}\norm{\grad^2 F_k(v)}\le L$.
\end{assumption}

We further assume $w$ lies in the \emph{interior} of $\mathcal B$. Since each
flow has speed $\norm{\dot\phi_k^s}=\norm{\grad F_k}\le G$ (and likewise
$\norm{\dot\psi^s}\le G$), there is a positive escape time
$T_0\ge \mathrm{dist}(w,\partial\mathcal B)/G$ such that, for every
$0\le T\le T_0$, all client trajectories $\phi_k^s$ \emph{and} the global
trajectory $\psi^s$ remain in $\mathcal B$ for $s\in[0,T]$. Define the
gradient-flow maps
\begin{equation}
\Phi_k^{T}(w)=\phi_k^{T},\quad \dot\phi_k^{s}=-\grad F_k(\phi_k^{s}),\ \phi_k^0=w;
\qquad
\Phi_f^{T}(w)=\psi^{T},\quad \dot\psi^{s}=-\grad f(\psi^{s}),\ \psi^0=w.
\label{eq:flows}
\end{equation}
The FedAvg \emph{server map} is the weighted average of per-client flow maps,
\begin{equation}
\mathcal S^{T}(w):=\sum_{k=1}^K\beta_k\,\Phi_k^{T}(w).
\label{eq:server}
\end{equation}
The horizon $T=\eta u_k$ is the continuous-time image of $u_k=En_k/B$ discrete
local steps of size $\eta$; see Lemma~\ref{lem:euler} for the discrete link.

\section{Lie--Taylor expansion of a gradient flow}

\begin{lemma}[Third-order flow expansion]\label{lem:lie}
Under Assumption~\ref{ass:smooth}, for $0\le T\le T_0$ (with $T_0$ chosen so the
trajectory stays in $\mathcal B$),
\begin{equation}
\Phi_k^{T}(w)=w-T\,g_k+\frac{T^2}{2}\,H_k g_k+R_k(T),
\qquad
\norm{R_k(T)}\le \frac{T^3}{6}\,C_3,
\label{eq:lie}
\end{equation}
where $C_3:=\rho G^2+L^2 G$ bounds the third $s$-derivative of $\phi_k^s$ on
$\mathcal B$. The same holds for $\Phi_f^{T}$ with $(g_k,H_k)$ replaced by
$(g,H)$ and the same constant (since $f$ is a convex combination of the $F_k$).
\end{lemma}

\begin{proof}
Let $V=-\grad F_k$, so $\dot\phi^s=V(\phi^s)$. Then
$\tfrac{d}{ds}\phi^s=V(\phi^s)$ and, differentiating again,
$\tfrac{d^2}{ds^2}\phi^s=DV(\phi^s)\,V(\phi^s)$, where $DV=-\grad^2F_k$. At $s=0$:
$\phi^0=w$, $\dot\phi^0=-g_k$, $\ddot\phi^0=(-H_k)(-g_k)=H_kg_k$. Taylor's
theorem with the Lagrange/integral remainder gives \eqref{eq:lie} with
$R_k(T)=\tfrac12\int_0^T (T-s)^2\,\dddot\phi^{\,s}\,ds$ in the integral form, so
$\norm{R_k(T)}\le \tfrac{T^3}{6}\sup_{s\le T}\norm{\dddot\phi^s}$. Differentiating
a third time,
\[
\dddot\phi^s=\frac{d}{ds}\big[DV(\phi^s)V(\phi^s)\big]
=D^2V(\phi^s)\big(V(\phi^s),V(\phi^s)\big)+DV(\phi^s)DV(\phi^s)V(\phi^s),
\]
and $\norm{D^2V}=\norm{\grad^3F_k}\le\rho$, $\norm{DV}=\norm{\grad^2F_k}\le L$,
$\norm{V}=\norm{\grad F_k}\le G$ on $\mathcal B$, so
$\norm{\dddot\phi^s}\le \rho G^2+L^2G=C_3$. The statement for $\Phi_f^T$ is
identical because $\grad^3 f=\sum_k\beta_k\grad^3F_k$ obeys the same bound by
the triangle inequality, using $\beta_k\ge0$ and $\sum_k\beta_k=1$ (no convexity
of the objectives is assumed or used anywhere).
\end{proof}

\section{Main result}

\begin{theorem}[Effective-ODE drift bound]\label{thm:main}
Let Assumption~\ref{ass:smooth} hold and suppose all clients share the same
horizon $T_k\equiv T$ (equal effective time; e.g.\ equal client sizes). Then for
$0\le T\le T_0$,
\begin{equation}
\mathcal S^{T}(w)-\Phi_f^{T}(w)=\frac{T^2}{2}\,\Gamma+R(T),
\qquad
\Gamma:=\sum_k\beta_k H_k g_k-Hg=\Cov_k(H_k,g_k),
\label{eq:thm}
\end{equation}
with $\norm{R(T)}\le \tfrac{T^3}{3}C_3$. Consequently, writing
$\sigma_{HG}:=\norm{\Gamma}$,
\begin{equation}
\Big|\;\norm{\mathcal S^{T}(w)-\Phi_f^{T}(w)}-\frac{T^2}{2}\,\sigma_{HG}\;\Big|
\;\le\;\frac{T^3}{3}\,C_3,
\qquad\text{i.e.}\qquad
\norm{\mathcal S^{T}(w)-\Phi_f^{T}(w)}=\frac{T^2}{2}\,\sigma_{HG}+O(T^3),
\label{eq:thm2}
\end{equation}
and the leading-order drift direction is exactly $\Gamma/\norm{\Gamma}$.
\end{theorem}

\begin{proof}
Average \eqref{eq:lie} over $k$ with weights $\beta_k$:
\[
\mathcal S^{T}(w)=\sum_k\beta_k\Phi_k^{T}(w)
=w-T\!\sum_k\beta_k g_k+\frac{T^2}{2}\sum_k\beta_k H_kg_k+\sum_k\beta_k R_k(T)
=w-Tg+\frac{T^2}{2}\sum_k\beta_kH_kg_k+R_1(T),
\]
with $\norm{R_1(T)}=\norm{\sum_k\beta_kR_k(T)}\le\sum_k\beta_k\tfrac{T^3}{6}C_3
=\tfrac{T^3}{6}C_3$ (Jensen / triangle inequality, $\sum_k\beta_k=1$). By
Lemma~\ref{lem:lie} applied to $f$,
$\Phi_f^{T}(w)=w-Tg+\tfrac{T^2}{2}Hg+R_2(T)$ with
$\norm{R_2(T)}\le\tfrac{T^3}{6}C_3$. Subtracting, the $O(1)$ term $w$ and the
$O(T)$ term $-Tg$ cancel exactly, leaving
\[
\mathcal S^{T}(w)-\Phi_f^{T}(w)
=\frac{T^2}{2}\Big(\sum_k\beta_kH_kg_k-Hg\Big)+\big(R_1(T)-R_2(T)\big)
=\frac{T^2}{2}\,\Gamma+R(T),
\]
$\norm{R(T)}\le\norm{R_1}+\norm{R_2}\le\tfrac{T^3}{3}C_3$. The covariance
identity is the algebra
\[
\Gamma=\sum_k\beta_kH_kg_k-\Big(\sum_k\beta_kH_k\Big)\Big(\sum_k\beta_kg_k\Big)
=\mathbb E_k[H_kg_k]-\mathbb E_k[H_k]\,\mathbb E_k[g_k]=\Cov_k(H_k,g_k),
\]
where $\mathbb E_k$ is expectation under the probability weights $\beta_k$.
Finally \eqref{eq:thm2} is the reverse triangle inequality applied to
\eqref{eq:thm}.
\end{proof}

\begin{remark}[Unequal horizons]\label{rem:unequal}
If client horizons differ, $T_k=\eta u_k$ with $u_k=En_k/B$, assume
$\max_k T_k\le T_0$ (so every $\phi_k^{T_k}$ and the reference $\psi^{\bar T}$
stay in $\mathcal B$), and define the weighted mean horizon
$\bar T=\sum_k\beta_k T_k$ and the heterogeneous server map
$\mathcal S_{\mathrm{het}}(w):=\sum_k\beta_k\Phi_k^{T_k}(w)$. Then
$\mathcal S_{\mathrm{het}}(w)-\Phi_f^{\bar T}(w)=-\sum_k\beta_k(T_k-\bar T)g_k
+\tfrac12\big(\sum_k\beta_kT_k^2H_kg_k-\bar T^2Hg\big)+O(T^3)$. The new
first-order term $-\sum_k\beta_k(T_k-\bar T)g_k=-\Cov_k(T_k,g_k)$ is the
\emph{size-heterogeneity} drift; it vanishes when client sizes are equal
($T_k\equiv T$, recovering Theorem~\ref{thm:main}) or when one compares against
the size-weighted reference $\sum_k\beta_k\Phi_f^{T_k}$. Either way the result is
governed entirely by the horizons $\{T_k=\eta E n_k/B\}$, never by $(E,B,\eta)$
separately.
\end{remark}

\begin{lemma}[Discrete-to-flow consistency]\label{lem:euler}
Let $\widehat\Phi_k^{T}$ be the $u_k$-step explicit Euler discretization of
\eqref{eq:flows} with step $\eta=T/u_k$ (i.e.\ literal full-batch local GD). Then
$\norm{\widehat\Phi_k^{T}(w)-\Phi_k^{T}(w)}\le \tfrac{\eta}{2}\,G\,(e^{LT}-1)
=\tfrac{GT(e^{LT}-1)}{2u_k}=O(\eta)\to0$ as $u_k\to\infty$ at fixed $T$. Hence the discrete
FedAvg server map satisfies the bound \eqref{eq:thm2} up to an additional
$O(\eta)=O(T/u_k)$ term, which the prototype confirms vanishes as $E$ grows at
fixed $T$.
\end{lemma}

\begin{proof}
Standard global-error bound for explicit Euler on a field with Jacobian bound
$L$ and second-derivative-of-solution bound $\norm{\ddot\phi}\le LG$ (since
$\ddot\phi=DV\,V$ with $\norm{DV}\le L$, $\norm{V}\le G$): the global error is
$\le \tfrac{\eta}{2L}(e^{LT}-1)\sup\norm{\ddot\phi}\le\tfrac{\eta}{2L}(e^{LT}-1)\,LG
=\tfrac{\eta}{2}G(e^{LT}-1)$, which is $O(\eta)=O(T/u_k)$ for $LT$ bounded.
Averaging over $k$ with $\sum_k\beta_k=1$ preserves the bound.
\end{proof}

\section{Consequences}

\paragraph{Effective outer ODE.} Iterating Theorem~\ref{thm:main}, the server
sequence $w_{t+1}=\Phi_f^{T}(w_t)-\tfrac{T^2}{2}\Gamma(w_t)+O(T^3)$ is, in the
slow time $s=tT$, a consistent one-step discretization of the modified
(backward-error) flow below: over a slow-time interval $[0,S]$ the $N=S/T$ steps
accumulate $N\cdot O(T^3)=O(T^2)$ error, so to first order in $T$ the iterates
track the perturbed flow
\[
\frac{dw}{ds}=-\grad f(w)-\frac{T}{2}\,\Cov_k\big(H_k,g_k\big)+O(T^2).
\]
The heterogeneity field $\Cov_k(H_k,g_k)$ is the continuous-time avatar of the
deep-dive's $\Delta$; it vanishes when clients are statistically identical
($H_k\equiv H$ or $g_k\equiv g$), recovering exact centralized gradient flow.

\paragraph{Recovering and correcting the heuristic.} For $E=2$, $B=\infty$ we
have $u_k=2$, $T=2\eta$, and Theorem~\ref{thm:main} gives drift
$\tfrac{(2\eta)^2}{2}\sigma_{HG}=2\eta^2\sigma_{HG}$ with the \emph{same} object
$\Gamma=\Cov_k(H_k,g_k)$ as the two-step Taylor sketch. This flow value
$2\eta^2$ is exactly twice the discrete two-step heuristic $\eta^2$; the factor
$2$ is the explicit-Euler global error of Lemma~\ref{lem:euler} at $u_k=2$, and it
shrinks as $O(1/u_k)$ once $E$ grows. For general $E$ the correct prefactor is
$T^2/2=(\eta u_k)^2/2$, not the frozen $\eta^2$: relative to the raw $\eta^2$
heuristic the magnitude grows by $u_k^2/2$ (a factor $2$ already at $E=2$ and
$12.5$ at $E=5$). The single invariant is $T=\eta E n_k/B$.

\section{Softmax cross-entropy satisfies Assumption~\ref{ass:smooth}}
For mean cross-entropy with softmax probabilities $p_i=\mathrm{softmax}(x_i^\top W)$,
the gradient is $\tfrac1m\sum_i x_i(p_i-y_i)^\top$ and the Hessian acts as
$H[V]=\tfrac1m\sum_i x_i\big((\diag(p_i)-p_ip_i^\top)(x_i^\top V)\big)^\top$
(used verbatim in the prototype's exact Hessian-vector product). All derivatives
are bounded on \emph{all} of $W$-space because $p\mapsto$ softmax is $C^\infty$
with bounded derivatives ($\norm{p}\le1$) and the data $\{x_i\}$ are fixed;
concretely $G\le\sqrt2\,\max_i\norm{x_i}$, $L\le\max_i\norm{x_i}^2$, and
$\rho\le 2\max_i\norm{x_i}^3$. Since these hold on every ball, one may take
$\mathcal B$ large enough around $w$ that the speed-$\le G$ flow stays inside for
$T\le T_0=\mathrm{dist}(w,\partial\mathcal B)/G>0$; hence
Assumption~\ref{ass:smooth} holds in full (both the constants and the
trajectory-confinement).
